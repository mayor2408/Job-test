Время: $O(N)$Память: $O(1)$Задание 3: ФакторизацияPythondef prime_factors(n: int) -> list:
    factors, d = [], 2
    while d * d <= n:
        while n % d == 0:
            factors.append(d)
            n //= d
        d += 1
    if n > 1:
        factors.append(n)
    return factors
